-- ini diterapkan setelah melakukan supabase-first-setup

-- Tambah kolom role dengan default value 'user'
ALTER TABLE public.profiles 
ADD COLUMN role TEXT 
DEFAULT 'user' 
CHECK (role IN ('user', 'admin'))
NOT NULL;

-- Update fungsi handle_new_user
CREATE OR REPLACE FUNCTION public.handle_new_user() 
RETURNS TRIGGER AS $$
BEGIN
    INSERT INTO public.profiles (id, username, full_name, email, avatar_url, role)
    VALUES (
        new.id,
        new.raw_user_meta_data->>'username',
        new.raw_user_meta_data->>'full_name',
        new.email,
        new.raw_user_meta_data->>'avatar_url',
        COALESCE(new.raw_user_meta_data->>'role', 'user')  -- Default ke 'user' jika tidak ada
    );
    RETURN new;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;
