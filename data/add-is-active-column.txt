-- Tambah kolom is_active ke table profiles
ALTER TABLE public.profiles 
ADD COLUMN is_active BOOLEAN DEFAULT true;

-- Update fungsi handle_new_user untuk menyertakan is_active
CREATE OR REPLACE FUNCTION public.handle_new_user() 
RETURNS TRIGGER AS $$
BEGIN
    INSERT INTO public.profiles (id, username, full_name, email, avatar_url, role, is_active)
    VALUES (
        new.id,
        new.raw_user_meta_data->>'username',
        new.raw_user_meta_data->>'full_name',
        new.email,
        new.raw_user_meta_data->>'avatar_url',
        COALESCE(new.raw_user_meta_data->>'role', 'user'),
        true  -- Set default active
    );
    RETURN new;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;
